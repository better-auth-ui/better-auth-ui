---
title: useSendVerificationEmail
description: Resend the verification email to an address.
---

## Usage

```tsx
import { authClient } from "@/lib/auth-client"
import { useSendVerificationEmail } from "@better-auth-ui/react"

const { mutate: sendVerificationEmail } = useSendVerificationEmail(authClient)

sendVerificationEmail({
  email: "alice@example.com",
  callbackURL: "/dashboard"
})
```

## Options factory

```tsx
import { sendVerificationEmailOptions } from "@better-auth-ui/core"
import { useMutation } from "@tanstack/react-query"

const { mutate } = useMutation(sendVerificationEmailOptions(authClient))
```
