---
title: useRequestPasswordReset
description: Send a password reset email for the forgot-password flow.
---

## Usage

```tsx
import { useAuth, useRequestPasswordReset } from "@better-auth-ui/react"

const { authClient } = useAuth()
const { mutate: requestPasswordReset } = useRequestPasswordReset(authClient)

requestPasswordReset({
  email: "alice@example.com",
  redirectTo: "/auth/reset-password"
})
```

## Options factory

```tsx
import { requestPasswordResetOptions } from "@better-auth-ui/core"
import { useMutation } from "@tanstack/react-query"

const { mutate } = useMutation(requestPasswordResetOptions(authClient))
```
