---
title: useDeletePasskey
description: Remove a passkey from the current user.
---

Refetches the passkey list on success.

## Usage

```tsx
import { useDeletePasskey } from "@better-auth-ui/react/plugins/passkey"
import { authClient } from "@/lib/auth-client"

const { mutate: deletePasskey } = useDeletePasskey(authClient)

deletePasskey({ id: passkey.id })
```

`useDeletePasskey` requires the Better Auth `passkey` plugin on your app auth client.

## Options factory

```tsx
import { deletePasskeyOptions } from "@better-auth-ui/core/plugins/passkey"
import { useMutation } from "@tanstack/react-query"

const { mutate } = useMutation(deletePasskeyOptions(authClient))
```
