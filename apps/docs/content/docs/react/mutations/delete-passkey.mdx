---
title: useDeletePasskey
description: Remove a passkey from the current user.
---

Refetches the passkey list on success.

## Usage

```tsx
import {
  type PasskeyAuthClient,
  useAuth,
  useDeletePasskey
} from "@better-auth-ui/react"

const { authClient } = useAuth<PasskeyAuthClient>()
const { mutate: deletePasskey } = useDeletePasskey(authClient)

deletePasskey({ id: passkey.id })
```

`useDeletePasskey` requires the Better Auth `passkey` plugin. Type `useAuth` with `PasskeyAuthClient` so TypeScript picks up the plugin-extended method signature.

## Options factory

```tsx
import { deletePasskeyOptions } from "@better-auth-ui/react"
import { useMutation } from "@tanstack/react-query"

const { mutate } = useMutation(deletePasskeyOptions(authClient))
```
