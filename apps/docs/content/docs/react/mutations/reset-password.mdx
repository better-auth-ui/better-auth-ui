---
title: useResetPassword
description: Complete the password reset flow with a token.
---

## Usage

```tsx
import { useAuth, useResetPassword } from "@better-auth-ui/react"

const { authClient } = useAuth()
const { mutate: resetPassword } = useResetPassword(authClient)

resetPassword({
  newPassword: "hunter3",
  token
})
```

## Options factory

```tsx
import { resetPasswordOptions } from "@better-auth-ui/core"
import { useMutation } from "@tanstack/react-query"

const { mutate } = useMutation(resetPasswordOptions(authClient))
```
