title: useChangeEmail
description: Change the current user's email address.
---

## Usage

```tsx
import { changeEmailOptions } from "@better-auth-ui/core"
import { useMutation } from "@tanstack/solid-query"

const changeEmail = useMutation(() => changeEmailOptions(authClient))

changeEmail.mutate({ newEmail: "alice@example.com" })
```
