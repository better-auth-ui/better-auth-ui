---
title: useSendVerificationEmail
description: Send an email verification message.
---

## Usage

```tsx
import { useSendVerificationEmail } from "@better-auth-ui/solid"

const sendVerificationEmail = useSendVerificationEmail(authClient)

sendVerificationEmail.mutate({
  email: "alice@example.com",
  callbackURL: "/dashboard"
})
```

## Options factory

```tsx
import { sendVerificationEmailOptions } from "@better-auth-ui/core"
import { createMutation } from "@tanstack/solid-query"

const sendVerificationEmail = createMutation(() =>
  sendVerificationEmailOptions(authClient)
)
```
