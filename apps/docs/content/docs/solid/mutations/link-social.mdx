title: useLinkSocial
description: Link a social provider to the current account.
---

## Usage

```tsx
import { linkSocialOptions } from "@better-auth-ui/core"
import { useMutation } from "@tanstack/solid-query"

const linkSocial = useMutation(() => linkSocialOptions(authClient))

linkSocial.mutate({ provider: "github" })
```
